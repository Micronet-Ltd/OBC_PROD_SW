CREATE TABLE IF NOT EXISTS system_results (
	test_id integer PRIMARY KEY,
	dt datetime default current_timestamp,
	test_version text,
	device_type text,
	serial text,
	imei text,
	imei_test boolean,
	serial_test boolean,
	mcu_ver text,
	fpga_ver text,
	os_ver text,
	version_test boolean,
	led_test boolean,
	sd_card_test boolean,
	wifi_rssi text,
	wifi_test boolean,
	canbus_test boolean,
	swc_test boolean,
	j1708_test boolean,
	com_test boolean,
	nfc_test boolean,
	help_key_test boolean,
	audio_test boolean,
	temperature_test boolean,
	rtc_test boolean,
	accelerometer_test boolean,
	gpio_test boolean,
	gpio_inputs_test boolean,
	wiggle_test boolean,
	supercap_test boolean,
	all_tests boolean
);

	fpga_ver text,

	version_test boolean,
	led_test boolean,

	swc_test boolean,

	help_key_test boolean,
	audio_test boolean,

	accelerometer_test boolean,
